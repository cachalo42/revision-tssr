Linux - Révisions TSSR

Gestion des utilisateurs

# Créer un utilisateur
sudo adduser dupont

# Supprimer un utilisateur
sudo deluser dupont

# Modifier le mot de passe
sudo passwd dupont

Gestion des dossiers et fichiers

# Créer un dossier
mkdir /home/dupont/docs

# Supprimer un dossier
rm -r /home/dupont/docs

# Déplacer un dossier
mv /home/dupont/docs /srv/docs

# Renommer un dossier
mv /home/dupont/docs /home/dupont/documents

# Créer un fichier
touch fichier.txt

# Supprimer un fichier
rm fichier.txt

Propriétaires et droits

# Modifier le propriétaire
sudo chown dupont:dupont /srv/docs

# Modifier les droits (lecture/écriture/exécution)
chmod 755 /srv/docs

# Droits avancés avec setfacl
setfacl -m u:dupont:rwx /srv/docs

Services et processus

# Vérifier l’état d’un service
systemctl status ssh

# Redémarrer un service
sudo systemctl restart apache2

# Activer un service au démarrage
sudo systemctl enable apache2

# Vérifier les processus
ps aux | grep apache

# Vérifier la RAM et le disque
free -h
df -h
