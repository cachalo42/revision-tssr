Linux - Révisions TSSR

Gestion des utilisateurs

# Créer un utilisateur
sudo adduser dupont

# Supprimer un utilisateur (sans supprimer son dossier home)
sudo deluser dupont

# Supprimer un utilisateur et son dossier home
sudo deluser --remove-home dupont

# Supprimer un utilisateur et tous ses fichiers
sudo deluser --remove-all-files dupont

# Modifier le mot de passe
sudo passwd dupont

Gestion des dossiers et fichiers

# Créer un dossier
mkdir /home/dupont/docs

# Supprimer un dossier vide
rmdir /home/dupont/docs

# Supprimer un fichier
rm fichier.txt

# Supprimer un dossier et son contenu
rm -r /home/dupont/docs

# Supprimer un dossier et son contenu sans confirmation (dangereux)
rm -rf /home/dupont/docs

# Déplacer un dossier
mv /home/dupont/docs /srv/docs

# Renommer un dossier
mv /home/dupont/docs /home/dupont/documents

# Créer un fichier
touch fichier.txt

Propriétaires et droits

# Modifier le propriétaire
sudo chown dupont fichier.txt

# Modifier propriétaire et groupe
sudo chown dupont:it fichier.txt

# Modifier récursivement sur un dossier
sudo chown -R dupont:it /srv/docs

# Modifier les droits (lecture/écriture/exécution)
chmod 644 fichier.txt   # propriétaire lecture/écriture, groupe+autres lecture
chmod 755 /srv/docs     # propriétaire rwx, groupe+autres rx

# Notation symbolique
chmod u+rwx,g+rx,o-r /srv/app

Services et processus

# Vérifier l’état d’un service
systemctl status ssh

# Redémarrer un service
sudo systemctl restart apache2

# Activer un service au démarrage
sudo systemctl enable apache2

# Vérifier les processus
ps aux | grep apache

# Vérifier la RAM et le disque
free -h
df -h

Commandes réseau utiles

# Afficher configuration IP
ip a

# Vérifier la table ARP
arp -a

# Tester la connectivité
ping 192.168.1.1

# Tracer la route
traceroute google.com
